% Provides tables for short reference materials

% Environments

% Commands
\newcommand{\titletable}[2]{
  \begin{center}
    {\huge #1} \\
    {\small #2} \\
  \end{center}
}

\newcommand{\refentry}[2]{
  \makebox[\linewidth]{
    \noindent\begin{minipage}[c][][c]{0.46\linewidth}
      \raggedright
      \texttt{ #1 }
    \end{minipage}
    \dotfill
    \begin{minipage}[c][][c]{0.53\hsize}
      \raggedleft
      { #2 }
    \end{minipage}
  }

  \vspace{1ex}
}
