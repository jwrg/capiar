%
% Keystroke-like font
%
% As per some available fonts which mimic the look of
% keys on a keyboard, except this this adjustable.
%

\newcommand*\keystroke[1]{%
  \tikz[baseline=(key.base)]
    \node[%
      draw,
      fill=white,
      drop shadow={shadow xshift=0.25ex,shadow yshift=-0.25ex,fill=black,opacity=0.75},
      rectangle,
      rounded corners=4pt,
      inner sep=6pt,
      minimum height=16pt,
      minimum width=30pt,
      line width=0.5pt,
      font=\large\sffamily\bfseries
    ](key) {#1\strut}
  ;
}
